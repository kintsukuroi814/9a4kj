Private Sub CB_Dodaj_Click()
'promjena gumbi
CB_Promjeni.Enabled = False
CB_Obriši.Enabled = False
'onemogucili smo promjenu i brisanje podataka za vrijeme dodavan
'provjera moda
If dodavanje = False Then
'brisanje okvira za tekst
Call Obrišiokvire
'promjena gumbi
CB_Dodaj.Enabled = False
Tiskanje.Enabled = False
'promjena moda
dodavanje = True
'otvaranje okvira za upis teksta
Call Otvoriokvire
'postavljanje kursora u prvi okvir
TB_Tekst(0).SetFocus
Exit Sub
End If
'povecavanje rednog broja i rezervne matrice
broj = ukupno + 1
ReDim rezerva$(broj, 6)
'prebacivanje podataka u rezervnu matricu
For n = 1 To ukupno
For m = 1 To 6
rezerva$(n, m) = podaci$(n, m)
Next m
Next n
'Slijedi prebacivanje svih podataka osoba u tako povecanu matricu rezerva$.
'zapis podataka iz okvira za tekst
For n = 1 To 6
rezerva$(broj, n) = TB_Tekst(n - 1).Text
Next n
'Podatke iz svih okvira za tekst zapisujemo kao zadnji podatak matrice,dakle
'na mjesto koje smo stvorili povecavanjem matrice u prethodnom bloku.

'povecanje ukupnog broja i glavne matrice
ukupno = ukupno + 1
ReDim podaci$(broj, 6)
'vracanje podataka u glavnu matricu
For n = 1 To ukupno
For m = 1 To 6
podaci$(n, m) = rezerva$(n, m)
Next m
Next n

'dodavanje imena i prezimena
LB_Popis.AddItem TB_Tekst(1).Text + "," + TB_Tekst(0).Text
LB_Popis.ItemData(LB_Popis.NewIndex) = broj

'brisanje sadržaja svih okvira za tekst
Call Obrišiokvire
'postavljanje novih vrijednosti
dodavanje = False
CB_Dodaj.Enabled = True
'zatvaranje okvira za upis teksta
Call Zatvoriokvire
End Sub

Private Sub CB_Obriši_Click()
'skidanje podataka sa liste
LB_Popis.RemoveItem indeks
'Prvo je potrebno obrisati ime i prezime sa liste.Parametar
'indeks dobio je svoju vrijednost odabirom podatka na listi i oznacuje
'redni broj podatka u ispisu.
'Nakon toga brišem podatke iz glavne matrice
For n = 1 To 6
podaci$(broj, n) = ""

'brisanje svih okvira za tekst
Call Obrišiokvire
'postavljanje gumbi
CB_Obriši.Enabled = False
CB_Dodaj.Enabled = True
'zatvaranje okvira za upis teksta
Call Zatvoriokvire
'onemogucavanje tiskanja
Tiskanje.Enabled = False

End Sub

Public Sub Ucitaj()
'kreiranje datoteke ako ne postoji
If Dir("C:\Adresar\Adresar.pri", 0) = "" Then
Open "C:\Adresar\Adresar.pri" For Output As #1
Write #1, 0
Close #1
ukupno = 0
End If

'otvori datoteku za citanje podataka
Open "C:\Adresar\Adresar.pri" For Input As #1

Input #1, ukupno
'ako nema podataka
If ukupno = 0 Then
Close #1
Exit Sub
End If
'povecavanje matrica prema broju podataka

ReDim podaci$(ukupno, 6)
ReDim rezerva$(ukupno, 6)
'citanje podataka

For n = 1 To ukupno
For m = 1 To 6
Input #1, a$
podaci$(n, m) = a$
Next m
LB_Popis.AddItem podaci$(n, 1) + " " + podaci$(n, 2)
LB_Popis.ItemData(LB_Popis.NewIndex) = n
Next n


Close #1

End Sub

Private Sub TB_Tekst_Change(Index As Integer)
'postoji li ime i prezime
If TB_Tekst(1).Text = "" Or TB_Tekst(0).Text = "" Then
savtekst = 0
Else
savtekst = 1
End If
'što ako postoji ili ne postoji
If savtekst > 0 Then
CB_Dodaj.Enabled = True
Else
CB_Dodaj.Enabled = False
CB_Promjeni.Enabled = False
End If
End Sub

Private Sub TB_Tekst_GotFocus(Index As Integer)
'promjena boja
TB_Tekst(Index).BackColor = RGB(0, 128, 128)
TB_Tekst(Index).ForeColor = RGB(255, 255, 0)
'podebljanje natpisa
L_Natpis(Index).FontBold = True
'postavljanje kursora na pocetak teksta
TB_Tekst(Index).SelStart = 0
'ako je odabran okvir sa napomenama
If Index = 5 Then
TB_Tekst(5).SelStart = Len(TB_Tekst(5).Text)
Exit Sub
End If
'odabir cijelog teksta u okviru


End Sub

Private Sub TB_Tekst_LostFocus(Index As Integer)
'promjena boja
TB_Tekst(Index).BackColor = RGB(255, 255, 255)
TB_Tekst(Index).ForeColor = RGB(0, 0, 0)
'normalni ispis natpisa
L_Natpis(Index).FontBold = False

End Sub

Public Sub Snimi()
'prebrojavanje punih zapisa
indeks = 0
For n = 1 To ukupno
If podaci$(n, 1) <> "" Then
indeks = indeks + 1
End If
Next n

'otvaranje datoteke za snimanje podataka
Open "C:\Adresar\Adresar.pri" For Output As #1
Write #1, indeks

'Slijedi otvaranje datoteke za zapisivanje podataka.U datoteku odmah upisujemo
'broj osoba.Ako nema podataka
If indeks = 0 Then
Close #1
Exit Sub
End If

'snimanje podataka
For n = 1 To ukupno
If podaci(n, 1) = "" Then GoTo nastavak
For m = 1 To 5
a$ = podaci$(n, m)
Write #1, a$;
Next m
Write #1, podaci$(n, 6)
nastavak:
Next n
Close #1

'brisanje okvira za tekst
Call Obrišiokvire
'svojstva gumbi
CB_Obriši.Enabled = False
CB_Dodaj.Enabled = True
Tiskanje.Enabled = False

End Sub
'Nakon snimanja svih podataka slijedi brisanje svih okvira za tekst
'omogucavanje dodavanja nove osobe te onemogucavanje brisanja i
'promjene jer ne postoje ispisani podaci koje bi se moglo obrisati ili promjeniti.

Public Sub Obrišiokvire()
'brisanje sadržaja svih okvira za tekst
For n = 0 To 5
TB_Tekst(n).Text = ""
Next n
End Sub

Public Sub Zatvoriokvire()
'zatvaranje okvira za tekst
For n = 0 To 5
TB_Tekst(n).Enabled = False
Next n

End Sub

Public Sub Otvoriokvire()
'otvaranje okvira za tekst
For n = 0 To 5
TB_Tekst(n).Enabled = True
Next n

End Sub
