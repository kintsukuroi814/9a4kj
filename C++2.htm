// Ovaj program upravlja komunikacijskim serijskim vratima bez prekidnih rutina

# include <dos.h>
# include <conio.h>
# include <stdio.h>
# define vrata 0x3f8

void setup (void)
{
outportb (vrata+1,0); // iskljuci prekide
outportb (vrata+3,0x80); // 7 bit LCR-a = 1 omoguci postavljanje djeljitelja
outportb (vrata+0,0x0c); // brzina prijenosa 9600 bps postavlja donji bajt
outportb (vrata+1,0x00); // brzina prijenosa 9600 bps postavlja gornji bajt
outportb (vrata+3,0x03); // postavljam parametre prijenosa 8N1
outportb (vrata+2,0xC7); // FIFO registar
}

void main (void)
{
int c ;
char ch;
setup();
printf("Jednostavan program u jeziku C za serijsku komunikaciju \n");
printf("Pritisni ESC za kraj ... \n\n");

do
{
c=inportb(vrata+5); //LSR Da li je pristigao kakav znak? Bit0=1 Da
if (c&1) // Ako je primljen znak...
{
ch=inportb(vrata); //Procitaj znak...
printf("%c",ch); //...pa ga ispisi na zaslon
}
if (kbhit()) //Ako je pritisnuta tipka na tipkovnici...
{
ch=getch(); // ...primi taj znak...
outportb(vrata,ch); // ...i posalji na serijska vrata...
}
}
while (ch!=27); //... dok promjenljiva ch nije 27 tj.ESC
}
